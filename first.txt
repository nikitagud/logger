package org.example;
import org.apache.log4j.Logger;
import java.util.Scanner;


class Main {
    public static void greeting() {
        Logger logger = Logger.getLogger(Main.class.getName());
        logger.info("Hello World");
    }

    public static void main(String[] args) {
        greeting();

        // isEven(args);

        the_largest(args);
    }

    public static void isEven(String[] args) {
        Scanner scanner = new Scanner(System.in);
        Logger logger = Logger.getLogger(Main.class.getName());
        logger.info("Enter the number:");
        int number = scanner.nextInt();

        if (number % 2 == 0) {
            logger.info(true);
        } else {
            logger.info(false);
        }

        scanner.close();




    }

    public static void the_largest(String[] args) {
        Scanner scanner = new Scanner(System.in);
        Logger logger = Logger.getLogger(Main.class.getName());
        logger.info("Enter the numbers:");
        int first_num = scanner.nextInt();
        int second_num = scanner.nextInt();

        if (first_num < second_num) {
            logger.info(second_num);
        } else if (first_num > second_num){
            logger.info(first_num);
        } else {
            logger.info("Numbers are equal.");
        }
        scanner.close();

    }

    public static void grades(String[] args) {
      Scanner scanner = new Scanner(System.in);
      Logger logger = Logger.getLogger(Main.class.getName());
      logger.info("Enter the number:");
      int grade = scanner.nextInt();
      if (grade >= 90) {
          logger.info("A");
      } else if (89 >= grade && grade >= 80) {
          logger.info("B");

      } else if (79 >= grade && grade >= 70) {
          logger.info("C");
      }
    }
}
