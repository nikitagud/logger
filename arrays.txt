package org.example;
import java.util.Arrays;
import java.util.Random;
import java.util.Scanner;

public class Second {

    // Method to fill array of length 10 with values from Scanner
    public static int[] fillArrayFromScanner(int length) {
        Scanner scanner = new Scanner(System.in);
        int[] array = new int[length];
        System.out.println("Enter " + length + " integer values:");
        for (int i = 0; i < length; i++) {
            array[i] = scanner.nextInt();
        }
        return array;
    }

    // Method to fill array of length 20 with random values (min: 88, max: 144)
    public static int[] fillArrayWithRandomValues(int length, int min, int max) {
        Random random = new Random();
        int[] array = new int[length];
        for (int i = 0; i < length; i++) {
            array[i] = random.nextInt(max - min + 1) + min;
        }
        return array;
    }

    // Method to sort array from min to max
    public static int[] sortArray(int[] array) {
        int[] sortedArray = array.clone(); // Cloning to avoid modifying the original array
        Arrays.sort(sortedArray);
        return sortedArray;
    }

    // Method to find the median of the array
    public static double findMedian(int[] array) {
        int[] sortedArray = sortArray(array);
        int length = sortedArray.length;
        if (length % 2 == 0) {
            return (sortedArray[length / 2 - 1] + sortedArray[length / 2]) / 2.0;
        } else {
            return sortedArray[length / 2];
        }
    }

    public static void main(String[] args) {
        // Array of length 10 with Scanner input
        int[] scannerArray = fillArrayFromScanner(10);
        System.out.println("Array from Scanner: " + Arrays.toString(scannerArray));

        // Array of length 20 with random values (88 to 144)
        int[] randomArray = fillArrayWithRandomValues(20, 88, 144);
        System.out.println("Random Array: " + Arrays.toString(randomArray));

        // Sorting and finding median
        int[] sortedArray = sortArray(randomArray);
        System.out.println("Sorted Random Array: " + Arrays.toString(sortedArray));

        double median = findMedian(randomArray);
        System.out.println("Median of Random Array: " + median);
    }
}
